% Create a lightweight command for the custom chapter title page
% Usage: \makecustomchapter{Title}{Content List}
\newcommand{\makecustomchapter}[2]{
    \cleardoublepage
    \thispagestyle{empty}
    \refstepcounter{chapter} % Properly increments the chapter number
    \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#1} % Adds to Table of Contents
    
    \vspace*{1.5in}
    \begin{center}
        {\Huge \textbf{\chaptertitlename\ \thechapter}} \\[\baselineskip]
        {\Huge \textbf{#1}} \\
        \vspace{2cm}
        {\small \textit{"Quote Quote Quote Quote Quote." Name}} \\
        \vspace{4cm}
        \begin{minipage}{0.8\textwidth}
            \centering
            \large\textbf{Contents}\\[0.5ex]
            \rule{\textwidth}{0.8pt}\\[1ex]
            \small #2 % This is where your custom content goes
        \end{minipage}
    \end{center}
    \vfill
    \clearpage
}
